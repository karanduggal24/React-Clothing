🚀 PROJECT OVERVIEW
================================================================================
A full-stack e-commerce clothing store built with React + Redux frontend and 
FastAPI + PostgreSQL backend. Features include user authentication, cart 
management, payment integration, admin panel, and responsive design.

Tech Stack:
- Frontend: React 19, Redux Toolkit, Tailwind CSS, Framer Motion
- Backend: FastAPI, SQLAlchemy, PostgreSQL
- Deployment: Vercel (Frontend), Render (Backend)
- Storage: Cloudinary (Images), Redis (Caching)
- Payment: Razorpay Integration

```javascript
// Cart migration on login
export const syncGuestCartToUser = createAsyncThunk(
  'cart/syncGuestCartToUser',
  async (userEmail, { getState }) => {
    const guestId = sessionStorage.getItem('cart_user_id');
    
    // Fetch guest cart items
    const guestItems = await fetchGuestCart(guestId);
    
    // Migrate each item to user cart
    for (const item of guestItems) {
      await addToUserCart(userEmail, item);
    }
    
    // Clean up guest cart
    await clearGuestCart(guestId);
    sessionStorage.removeItem('cart_user_id');
    
    return await fetchUserCart(userEmail);
  }
);
```

```javascript
// Real-time total calculation with validation
const recalculateTotals = (state) => {
  state.totalItems = state.items.reduce((total, item) => 
    total + item.quantity, 0);
  state.totalPrice = state.items.reduce((total, item) => 
    total + (item.price * item.quantity), 0);
};

// Stock validation on every cart operation
const addToCart = (state, action) => {
  const product = action.payload;
  const existingItem = state.items.find(item => item.id === product.id);
  
  if (existingItem) {
    if (existingItem.quantity >= product.stockQuantity) {
      toast.warning(`${product.name} reached stock limit!`);
      return; // Prevent adding more
    }
    existingItem.quantity += 1;
  }
  
  recalculateTotals(state);
};
```

```javascript
// Controlled inputs with real-time validation
const Login = () => {
  const [formData, setFormData] = useState({
    email: '',
    password: ''
  });
  const [errors, setErrors] = useState({});
  const [loading, setLoading] = useState(false);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
    
    // Clear error when user starts typing
    if (errors[name]) {
      setErrors(prev => ({
        ...prev,
        [name]: ''
      }));
    }
  };

  const validateForm = () => {
    const newErrors = {};
    
    if (!formData.email) {
      newErrors.email = 'Email is required';
    } else if (!/\S+@\S+\.\S+/.test(formData.email)) {
      newErrors.email = 'Email is invalid';
    }
    
    if (!formData.password) {
      newErrors.password = 'Password is required';
    } else if (formData.password.length < 6) {
      newErrors.password = 'Password must be at least 6 characters';
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    if (!validateForm()) return;
    
    setLoading(true);
    try {
      await dispatch(loginUser(formData)).unwrap();
      navigate('/');
      toast.success('Login successful!');
    } catch (error) {
      toast.error(error.message || 'Login failed');
    } finally {
      setLoading(false);
    }
  };
};
```

```javascript
// Complex form with file upload
const ProductForm = () => {
  const [productData, setProductData] = useState({
    name: '', price: '', category: '', quantity: '', image: ''
  });
  const [imagePreview, setImagePreview] = useState('');
  const [uploadingImage, setUploadingImage] = useState(false);

  const handleImageChange = async (event) => {
    const file = event.target.files[0];
    
    // Client-side validation
    const allowedTypes = ['image/jpeg', 'image/jpg', 'image/png', 'image/gif'];
    if (!allowedTypes.includes(file.type)) {
      toast.error('Invalid file type');
      return;
    }
    
    const maxSize = 5 * 1024 * 1024; // 5MB
    if (file.size > maxSize) {
      toast.error('File size too large');
      return;
    }

    // Show preview immediately
    const reader = new FileReader();
    reader.onload = (e) => setImagePreview(e.target.result);
    reader.readAsDataURL(file);

    // Upload to backend
    setUploadingImage(true);
    try {
      const formData = new FormData();
      formData.append('file', file);
      
      const response = await fetch('/api/products/upload-image', {
        method: 'POST',
        body: formData
      });
      
      const data = await response.json();
      setProductData(prev => ({ ...prev, image: data.path }));
      toast.success('Image uploaded successfully!');
    } catch (error) {
      toast.error('Failed to upload image');
    } finally {
      setUploadingImage(false);
    }
  };
};
```

🔥 THE HARDEST PART
================================================================================
CHALLENGE: Cart synchronization between guest and authenticated users

PROBLEM COMPLEXITY:
The most challenging feature was implementing seamless cart synchronization 
between guest users and authenticated users. This involved:

1. Guest users can add items to cart without account
2. Cart must persist across browser sessions
3. On login, merge guest cart with user's existing cart
4. Handle conflicts when same item exists in both carts
5. Maintain cart state during network failures
6. Ensure data consistency between frontend and backend

TECHNICAL CHALLENGES:
- Race conditions during cart sync
- Handling duplicate items with different quantities
- Managing session IDs for guest users
- Optimistic updates vs backend reality
- Error recovery and rollback mechanisms

SOLUTION IMPLEMENTATION:
```javascript
// Complex async thunk for cart migration
export const syncGuestCartToUser = createAsyncThunk(
  'cart/syncGuestCartToUser',
  async (userEmail, { rejectWithValue, getState }) => {
    try {
      const guestId = sessionStorage.getItem('cart_user_id');
      
      // If no guest cart, just fetch user cart
      if (!guestId) {
        const response = await fetch(`/api/cart/?user_id=${userEmail}`);
        return await response.json();
      }
      
      // Fetch both carts
      const [guestResponse, userResponse] = await Promise.all([
        fetch(`/api/cart/?user_id=${guestId}`),
        fetch(`/api/cart/?user_id=${userEmail}`)
      ]);
      
      const guestItems = await guestResponse.json();
      const userItems = await userResponse.json();
      
      // Merge logic: handle conflicts intelligently
      for (const guestItem of guestItems) {
        const existingUserItem = userItems.find(
          item => item.product_id === guestItem.product_id
        );
        
        if (existingUserItem) {
          // Combine quantities, respect stock limits
          const newQuantity = Math.min(
            existingUserItem.quantity + guestItem.quantity,
            guestItem.stock_quantity
          );
          
          await fetch(`/api/cart/${existingUserItem.id}`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ quantity: newQuantity })
          });
        } else {
          // Add new item to user cart
          await fetch('/api/cart/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              user_id: userEmail,
              product_id: guestItem.product_id,
              quantity: guestItem.quantity,
              // ... other item data
            })
          });
        }
      }
      
      // Clean up guest cart
      await fetch(`/api/cart/user/${guestId}`, { method: 'DELETE' });
      sessionStorage.removeItem('cart_user_id');
      
      // Fetch final merged cart
      const finalResponse = await fetch(`/api/cart/?user_id=${userEmail}`);
      return await finalResponse.json();
      
    } catch (error) {
      console.error('Cart sync error:', error);
      return rejectWithValue(error.message);
    }
  }
);
```

```javascript
// Server-side pagination with infinite scroll
const ProductsList = () => {
  const [page, setPage] = useState(1);
  const [products, setProducts] = useState([]);
  const [hasMore, setHasMore] = useState(true);
  const [loading, setLoading] = useState(false);

  const loadMoreProducts = useCallback(async () => {

    setLoading(true);
    try {
      const response = await fetch(

      );
      const newProducts = await response.json();
      
      if (newProducts.length < 20) setHasMore(false);
      
      setProducts(prev => [...prev, ...newProducts]);
      setPage(prev => prev + 1);
    } catch (error) {
      console.error('Failed to load products:', error);
    } finally {
      setLoading(false);
    }
  }, [page, selectedCategory, loading, hasMore]);

  // Intersection Observer for infinite scroll
  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && hasMore && !loading) {
          loadMoreProducts();
        }
      },
      { threshold: 0.1 }
    );

    const sentinel = document.getElementById('scroll-sentinel');
    if (sentinel) observer.observe(sentinel);

    return () => observer.disconnect();
  }, [loadMoreProducts, hasMore, loading]);
};
```

```python
redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cache_result(ttl_seconds=300):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # Generate cache key from function name and arguments
            cache_key = f"{func.__name__}:{hash(str(args) + str(kwargs))}"
            
            # Try cache first
            cached_result = redis_client.get(cache_key)
            if cached_result:
                return json.loads(cached_result)
            
            # Execute function and cache result
            result = await func(*args, **kwargs)
            redis_client.setex(
                cache_key, 
                ttl_seconds, 
                json.dumps(result, default=str)
            )
            
            return result
        return wrapper
    return decorator
```

```python
# Elasticsearch for advanced search
from elasticsearch import Elasticsearch

es = Elasticsearch([{'host': 'localhost', 'port': 9200}])

# Index products for search
def index_product(product):
    doc = {
        'name': product.name,
        'description': product.description,
        'category': product.category,
        'price': product.price,
        'tags': product.tags,
        'created_at': product.created_at
    }
    
    es.index(
        index='products',
        id=product.id,
        body=doc
    )

# Advanced search with filters and facets
def search_products(query, filters=None, page=1, size=20):
    body = {
        'query': {
            'bool': {
                'must': [
                    {
                        'multi_match': {
                            'query': query,
                            'fields': ['name^2', 'description', 'tags'],
                            'fuzziness': 'AUTO'
                        }
                    }
                ],
                'filter': []
            }
        },
        'aggs': {
            'categories': {
                'terms': {'field': 'category.keyword'}
            },
            'price_ranges': {
                'range': {
                    'field': 'price',
                    'ranges': [
                        {'to': 1000},
                        {'from': 1000, 'to': 5000},
                        {'from': 5000}
                    ]
                }
            }
        },
        'from': (page - 1) * size,
        'size': size
    }
    
    # Add filters
    if filters:
        if filters.get('category'):
            body['query']['bool']['filter'].append({
                'term': {'category.keyword': filters['category']}
            })
        
        if filters.get('price_min') or filters.get('price_max'):
            price_filter = {'range': {'price': {}}}
            if filters.get('price_min'):
                price_filter['range']['price']['gte'] = filters['price_min']
            if filters.get('price_max'):
                price_filter['range']['price']['lte'] = filters['price_max']
            body['query']['bool']['filter'].append(price_filter)
    
    return es.search(index='products', body=body)
```

```javascript
const ImageComponent = ({ src, alt, sizes }) => {
```
